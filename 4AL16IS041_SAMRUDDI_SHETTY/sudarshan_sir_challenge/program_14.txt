int divisibleSumPairs(int n, int k, int ar_size, int* ar) {
    // Complete this function
    int i,j;
    int counter=0;
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            if(i<j && (ar[i]+ar[j])%k==0){
                counter++;
            }
        }
    }
    return counter;
}
