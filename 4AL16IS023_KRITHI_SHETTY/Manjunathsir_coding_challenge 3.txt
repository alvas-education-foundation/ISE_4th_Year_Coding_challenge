  int findStep(int n) 

{ 

    if (n == 1 || n == 0) 

        return 1; 

    else if (n == 2) 

        return 2; 

  

    else

        return findStep(n - 3) + findStep(n - 2) + findStep(n - 1); 

} 

  

int main() 

{ 

    int n = 4; 

    printf("%d\n", findStep(n)); 

    return 0; 

}

int getMissingNo(int a[], int n) 

{ 

    int i, total; 

    total = (n + 1) * (n + 2) / 2; 

    for (i = 0; i < n; i++) 

        total -= a[i]; 

    return total; 

} 

  

int main() 

{
